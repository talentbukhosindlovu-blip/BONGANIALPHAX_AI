class InstitutionalSignalProcessor:
    def __init__(self, watchlist: List[str]):
        self.cached_garch_vols: Dict[str, float] = {}
        self.last_garch_times: Dict[str, datetime] = {}
        for symbol in watchlist:
            self.cached_garch_vols[symbol] = (
                0.550 if "BTC" in symbol else
                (0.224 if "XAU" in symbol else 0.125)
            )
            self.last_garch_times[symbol] = datetime.min.replace(
                tzinfo=timezone.utc
            )

    @staticmethod
    def execute_kalman_filter(
        observed_mid: float,
        state_x: float,
        covariance_P: float
    ) -> Tuple[float, float]:
        """Isolates true asset price vectors from market microstructure variance."""
        if math.isnan(observed_mid) or observed_mid <= 0:
            return state_x, covariance_P
        if state_x == 0.0:
            return observed_mid, 1.0

        prior_x, prior_P = state_x, covariance_P + 1e-6
        innovation = observed_mid - prior_x
        innovation_cov = prior_P + 5e-3
        kalman_gain = prior_P / innovation_cov

        return (
            float(prior_x + (kalman_gain * innovation)),
            float((1.0 - kalman_gain) * prior_P)
        )

    def compute_mle_garch_volatility(
        self,
        symbol: str,
        historical_prices: List[float]
    ) -> float:
        """Executes a background MLE solver on a 1-hour cache layer to prevent execution lag."""
        now = datetime.now(timezone.utc)

        if (now - self.last_garch_times[symbol]).total_seconds() < 3600:
            return self.cached_garch_vols[symbol]

        n_samples = len(historical_prices)
        if n_samples < 30:
            return self.cached_garch_vols[symbol]

        returns = np.diff(np.log(np.array(historical_prices)))
        returns = returns - np.mean(returns)

        def _garch_nll_objective(params: np.ndarray) -> float:
            omega, alpha, beta = params
            penalty = 0.0

            if omega <= 1e-9:
                penalty += (1e-9 - omega) * 1e7
                omega = 1e-9

            if alpha < 0.0:
                penalty += abs(alpha) * 1e7
                alpha = 0.0

            if beta < 0.0:
                penalty += abs(beta) * 1e7
                beta = 0.0

            if (alpha + beta) >= 0.999:
                excess = (alpha + beta) - 0.999
                penalty += excess * 1e7
                scale = 0.999 / (alpha + beta)
                alpha *= scale
                beta *= scale

            t = len(returns)
            variance = np.zeros(t)
            variance = np.var(returns)

            for i in range(1, t):
                variance[i] = (
                    omega +
                    alpha * (returns[i - 1] ** 2) +
                    beta * variance[i - 1]
                )
                if variance[i] <= 0:
                    variance[i] = 1e-9

            neg_log_likelihood = -0.5 * np.sum(
                np.log(2 * np.pi * variance) +
                (returns ** 2) / variance
            )

            return float(-neg_log_likelihood + penalty)

        try:
            res = optimize.minimize(
                _garch_nll_objective,
                np.array([1e-6, 0.05, 0.90]),
                bounds=(
                    (1e-9, 1e-4),
                    (1e-4, 0.25),
                    (0.70, 0.98)
                ),
                method="L-BFGS-B"
            )

            if res.success:
                self.cached_garch_vols[symbol] = float(
                    np.sqrt(
                        max(
                            1e-8,
                            res.x +
                            res.x * (returns[-1] ** 2) +
                            res.x * np.var(returns)
                        )
                    ) * np.sqrt(252)
                )
                self.last_garch_times[symbol] = now

        except Exception:
            pass

        return self.cached_garch_vols[symbol]

class QuantitativeEngineA:
    @staticmethod
    def evaluate_signal(
        bid: float,
        smoothed_mean: float,
        symbol: str
    ) -> str:
        threshold_pct = (
            0.0050 if "BTC" in symbol else
            (0.0012 if "XAU" in symbol else 0.0004)
        )

        upper_barrier = smoothed_mean * (1.0 + threshold_pct)
        lower_barrier = smoothed_mean * (1.0 - threshold_pct)

        if bid > upper_barrier:
            return "SELL"
        elif bid < lower_barrier:
            return "BUY"

        return "HOLD"


class QuantitativeEngineB:
    @staticmethod
    def evaluate_signal(price_history: List[float]) -> str:
        if len(price_history) < 10:
            return "HOLD"

        short_sma = np.mean(price_history[-5:])
        long_sma = np.mean(price_history[-10:])

        return (
            "BUY" if short_sma > long_sma
            else ("SELL" if short_sma < long_sma else "HOLD")
        )
